\newcommand*{\tauMem}{\gls{tauMem}\xspace}
\newglossaryentry{tauMem}{
	name=\ensuremath{\tau_\mathrm{m}},
	sort=tauMem,
	description={Membrane time constant, sometimes also referred to as $\tau_\mathrm{RC}$ (due to the membrane forming a resistor-capacitance circuit). In general, this is the time required to discharge a capacitor of capacitance $C$ over a resistor of value $R$ from an initial potential $u_0$ to the voltage $u_0 e^{-1}$. Conveniently, $\tau_\mathrm{RC}$ is just the product between the resistance and the capacitance, i.e., $\tau_\mathrm{RC} = RC$. We use this symbol to denote the membrane time-constant of a point neuron or a neural compartment. In this case, $\tau_\mathrm{m}$ is just given as $\tau_\mathrm{m} = C_\mathrm{m} g_\mathrm{L}^{-1}$}
}
